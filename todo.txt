import subprocess

def get_user_home(username):
    try:
        command = f"grep '^{username}:' /etc/passwd | cut -d':' -f6"
        process = subprocess.Popen(command, shell=True, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
        output, error = process.communicate()

        if output:
            home_directory = output.decode().strip()
            print(f"Username: {username}\tHome Directory: {home_directory}")
        else:
            print(f"User '{username}' not found.")
    except subprocess.CalledProcessError as e:
        print(f"Error occurred: {e.stderr}")

# Example usage
target_username = "john"
get_user_home(target_username)

import subprocess

def fetch_environment_variables():
    try:
        command = "printenv"
        process = subprocess.Popen(command, shell=True, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
        output, error = process.communicate()

        if output:
            env_variables = output.decode().splitlines()
            env_dict = {}
            for variable in env_variables:
                key, value = variable.split("=", 1)
                env_dict[key] = value
            return env_dict
        else:
            print("No environment variables found.")
    except subprocess.CalledProcessError as e:
        print(f"Error occurred: {e.stderr}")

# Example usage
environment_vars = fetch_environment_variables()
print(environment_vars)
